# 常用软件包安装

```bash
sudo apt install apt-transport-https aptitude
sudo apt install build-essential checkinstall cmake clang libreadline-dev libssl-dev libxml2-dev maven golang-go
sudo apt install zsh bat fzf exa git subversion
sudo apt install gnupg2 gpgv2 tree htop p7zip-full p7zip-rar neofetch ffmpeg axel mariadb-client
sudo apt install mariadb-server micro
sudo apt install fonts-noto-cjk fonts-noto-cjk-extra fonts-wqy-zenhei fonts-wqy-microhei language-pack-zh-hans language-pack-zh-hans-base manpages-zh xfonts-intl-chinese xfonts-intl-chinese-big xfonts-wqy fonts-noto-color-emoji
```
